7 void Motor_init(void) {
8  // Configure timer 3 (for motor A) and timer 4 (for motor B)
9  // to produce PWM signals on OC3A/OC3B and OC4A/OC4B with a
10  // frequency of 488.759Hz in phase correct PWM mode.
11  // Duty cycle is defined by OC4A/OC4B and OC3A/OC3B, respectively.
12  // OC3A/OC3B and OC4A/OC4B are connected to the motor driver TI DRV8833
13  // on pins AIN1/AIN2 and BIN2/BIN1, respectively.
14 
15  // Due to the phase correct PWM modes, updated values of OC3A/OC3B and
16  // OC4A/OC4B are buffered and taken over only once during a PWM period
17  // when the counter reaches TOP (0x1FFF).
18 
19  // DRV8833 requires two inputs for controlling a single motor with the
20  // following logic:
21  //
22  // A/BIN1 | A/BIN2 | A/BOUT1 | A/BOUT2 | Motor A/B
23  // -----------------------------------------------
24  // Low | Low | Z | Z | Coast/Bridge disabled (fast decay)
25  // High | Low | High | Low | Forward
26  // Low | High | Low | High | Reverse
27  // High | High | Low | Low | Break/Motor shorted (slow decay)
28  //
29  // For controlling a motor with a PWM signal, the slow-decay mode is used
30  // since this provides a quite linear relationship between PWM duty cycle
31  // and motor speed (under constant supply voltage and constant load conditions).
32  // The input pins are driven as follows:
33  //
34  // A/BIN1 | A/BIN2 | Motor A/B
35  // -----------------------------------------------
36  // High | PWM | Forward
37  // PWM | High | Reverse
38  //
39  // For stopping the motor, the fast-decay mode is chosen (Low on both inputs)
40  // as this disables the H-bridge and the outputs A/BOUT1 and A/BOUT2 are of
41  // high impedance.
42  // Thus, stopping a motor in fast-decay mode with inputs both low is a difference
43  // to stopping the motor in slow-decay mode with one input held high and the other
44  // with a PWM signal of 100% duty cycle.
45  //
46  // Speed of motor is influenced by duty cycle of PWM signals but may change
47  // due to variation in supply voltage and motor load.
48 
49  // disable power reduction of timers 3 and 4
50  PRR1 &= ~(_BV(PRTIM3) | _BV(PRTIM4));
51 
52  // stop timers 3 and 4 by selecting no clock
53  TCCR3B = 0;
54  TCCR4B = 0;
55 
56  // set timers to phase correct PWM mode (dual slope operation)
57  // ICRn defines TOP value (max value of counter) and is set to 0x1FFF (13 bits)
58  // use prescaler of N=1 => timer frequency equals CPU clock of 8MHz
59  // PWM frequency: 8MHz / (2*N*TOP) = 488.759Hz
60 
61  // WGMn1 (in TCCRnA) and WGMn3 (in TCCRnB): phase correct PWM mode (dual slope operation) with TOP defined by ICRn
62  // COMnA1, COMnA0: set OCnA on compare match with OCRnA when up-counting, clear OCnA on compare match with OCRnA when down-counting
63  // COMnB1, COMnB0: set OCnB on compare match with OCRnB when up-counting, clear OCnB on compare match with OCRnB when down-counting
64  TCCR3A = _BV(COM3A1) | _BV(COM3A0) | _BV(COM3B1) | _BV(COM3B0) | _BV(WGM31);
65  TCCR4A = _BV(COM4A1) | _BV(COM4A0) | _BV(COM4B1) | _BV(COM4B0) | _BV(WGM41);
66 
67  // no force output compare
68  TCCR3C = 0;
69  TCCR4C = 0;
70 
71  ATOMIC_BLOCK(ATOMIC_RESTORESTATE) {
72  // disable interrupts just in case because of TEMP register used by TCNTn, OCRnA and ICRn, see Section 17.3 of datasheet
73 
74  // set counter values to zero
75  TCNT3 = 0;
76  TCNT4 = 0;
77 
78  // set TOP to 0x1FFF (8191)
79  ICR3 = 0x1FFF;
80  ICR4 = 0x1FFF;
81 
82  // set OCRnA and OCRnB to TOP in order to have motors stopped in fast-decay mode after timers will be enabled
83  OCR3A = 0x1FFF;
84  OCR3B = 0x1FFF;
85  OCR4A = 0x1FFF;
86  OCR4B = 0x1FFF;
87 
88  // configure OC3A (PE3, AIN1) and OC3B (PE4, AIN2) as output for motor A
89  DDRE |= _BV(DDE3) | _BV(DDE4);
90  // configure OC4A (PH3, BIN2) and OC4B (PH4, BIN1) as output for motor B
91  DDRH |= _BV(DDH3) | _BV(DDH4);
92 
93  // Selecting the clock starts the timer and generation of PWM signals.
94  // A prescaler reset is performed while the clock sources are set.
95  // Disabling interrupts during timer activation ensures that the prescaler
96  // reset takes the smallest possible time as all timers except timer 2 are
97  // affected. Then, the prescalers are activated simultaneously, so that
98  // both timers start at the same time and updated values of the OCR registers
99  // are taken over simultaneously during normal operation.
100 
101  // Perform prescaler reset
102  GTCCR = _BV(TSM) | _BV(PSRSYNC);
103  // Select clock sources and prescaler
104  // CSn0: prescaler 1
105  // WGMn3: phase correct PWM mode with TOP defined by ICRn
106  TCCR3B = _BV(CS30) | _BV(WGM33);
107  TCCR4B = _BV(CS40) | _BV(WGM43);
108  // Enable prescaler, timers start running simultaneously
109  GTCCR = 0;
110  }
111 }

114 __attribute__ ((naked)) void Motor_setPWM_A(const int16_t pwm) {
115  asm volatile (
116  "sbrc %B[pwm], 7" "\n\t" // if pwm < 0
117  "rjmp L_pwmneg%=" "\n\t" // then goto L_pwmneg
118  "andi %B[pwm], 0x1F" "\n\t" // pwm & 0x1FFF
119  "in __tmp_reg__, __SREG__" "\n\t" // save status register in temporary register
120  "cli" "\n\t" // disable interrupts
121  "sts %B[POCR3A], __zero_reg__" "\n\t" // OCR3AH = 0, high byte first (OC3A high, AIN1)
122  "sts %A[POCR3A], __zero_reg__" "\n\t" // OCR3AL = 0, then low byte (OC3A high, AIN1)
123  "sts %B[POCR3B], %B[pwm]" "\n\t" // OCR3B = pwm & 0x1FFF, high byte first
124  "sts %A[POCR3B], %A[pwm]" "\n\t" // OCR3B = pwm & 0x1FFF, then low byte
125  "out __SREG__, __tmp_reg__" "\n\t" // restore status register
126  "ret" "\n\t" // return
127  "L_pwmneg%=:" "\n\t"
128  "neg %B[pwm]" "\n\t" // two's complement for high byte of pwm
129  "neg %A[pwm]" "\n\t" // two's complement for low byte of pwm
130  "sbc %B[pwm], __zero_reg__" "\n\t" // subtract 1 from high byte, if carry flag is set
131  "andi %B[pwm], 0x1F" "\n\t" // (-pwm) & 0x1FFF
132  "in __tmp_reg__, __SREG__" "\n\t" // save status register in temporary register
133  "cli" "\n\t" // disable interrupts
134  "sts %B[POCR3B], __zero_reg__" "\n\t" // OCR3BH = 0, high byte first (OC3B high, AIN2)
135  "sts %A[POCR3B], __zero_reg__" "\n\t" // OCR3BL = 0, then low byte (OC3B high, AIN2)
136  "sts %B[POCR3A], %B[pwm]" "\n\t" // OCR3A = (-pwm) & 0x1FFF, high byte first
137  "sts %A[POCR3A], %A[pwm]" "\n\t" // OCR3A = (-pwm) & 0x1FFF, then low byte
138  "out __SREG__, __tmp_reg__" "\n\t" // restore status register
139  "ret" "\n\t" // return
140  : // no outputs
141  : [POCR3A] "n" (& OCR3A),
142  [POCR3B] "n" (& OCR3B),
143  [pwm] "d" (pwm)
144  : // no clobber
145  );
