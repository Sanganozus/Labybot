134 void communication_writePacket(const Channel_t channel, const uint8_t* packet, const uint16_t size) {
135  // while writing each byte, the global checksum is calculated over the whole
136  // transmitted data including the header information
137 
138  // transmit low byte of payload size
139  register uint8_t chksum = (uint8_t)size;
140  uart_writeEscaped(chksum);
141 
142  // transmit high byte of payload size
143  register uint8_t byte = (uint8_t)(size >> 8);
144  uart_writeEscaped(byte);
145  chksum ^= byte;
146 
147  // compute 4-bit checksum of size and place it in high nibble,
148  // place channel number in low nibble and transmit
149  byte = (((chksum << 4) & 0xFF) ^ (chksum & 0xF0)) | (channel & 0x0F);
150  uart_writeEscaped(byte);
151  chksum ^= byte;
152 
153  // transmit packet payload and update checksum
154  for (uint16_t i = 0; i < size; i++) {
155  register uint8_t tmp = packet[i];
156  uart_writeEscaped(tmp);
157  chksum ^= tmp;
158  }
159 
160  // transmit checksum
161  uart_writeEscaped(chksum);
162  // transmit packet delimiter
163  uart_write(DELIM);
164 }

167 static __attribute__ ((noinline)) void readPackets(void) {
168  register uint8_t tmpChksum = inChksum;
169  register uint16_t tmpBufLen = inBufLen;
170  register uint8_t* tmpBuf = inBuf + tmpBufLen;
171  register uint8_t tmpIsESC = isESC;
172 
173  // while some data is available in the UART RX buffer
174  do {
175  // get the data byte
176  register uint8_t data = uart_read();
177 
178  if (tmpIsESC) { // if last data byte was the escape byte
179  // no more escaping
180  tmpIsESC = false;
181  } else if (data == ESC) { // if data byte is the escape byte
182  tmpIsESC = true; // set flag for indicating an escape sequence
183  continue;
184  } else if (data == DELIM) { // if data byte is delimiter, we should have a complete packet in the incoming buffer
185  if (tmpBufLen >= 4) { // if number of bytes in buffer is at least the minimum size, we might have received a full packet
186 
187  register uint8_t chksumSize = inBuf[0]; // read low byte of payload size
188  data = inBuf[1]; // read high byte of payload size
189  uint16_t size = chksumSize | ((uint16_t)data << 8); // compute payload size
190 
191  // calculate 4-bit checksum for payload size in low nibble
192  chksumSize ^= data;
193  chksumSize = (chksumSize >> 4) ^ (chksumSize & 0x0F);
194 
195  data = inBuf[2]; // read checksum for payload size (bits 7-4) and channel ID (bits 3-0)
196 
197  if ((data >> 4) == chksumSize) { // check for mismatch of payload size checksum
198  if (size == tmpBufLen - 4) { // check packet length
199  if (tmpChksum == 0) { // if global checksum is ok
200  register uint8_t channel = data & 0x0F; // get channel number
201  // execute callback function
202  if (communication_ChannelReceivers[channel])
203  (communication_ChannelReceivers[channel])(inBuf+3, size);
204  else
205  errors |= COMM_ERR_UNREGISTEREDCHANNEL;
206  } else
207  errors |= COMM_ERR_CHECKSUM;
208  } else
209  errors |= COMM_ERR_SIZE_MISMATCH;
210  } else
211  errors |= COMM_ERR_HEADER_CHECKSUM;
212  } else
213  errors |= COMM_ERR_TOO_SMALL;
214 
215  // clear incoming packet buffer and checksum
216  tmpBufLen = 0;
217  tmpChksum = 0;
218  tmpBuf = inBuf;
219  continue;
220  }
221 
222  // place byte in incoming packet buffer
223  *tmpBuf++ = data;
224  tmpChksum ^= data;
225 
226  if (++tmpBufLen == COMM_RECV_BUFFER_SIZE) { // if incoming packet buffer is full
227  // clear buffer
228  tmpBufLen = 0;
229  tmpChksum = 0;
230  tmpBuf = inBuf;
231  // set error flag
232  errors |= COMM_ERR_BUFFERFULL;
233  }
234  } while (uart_available());
235 
236  // update current checksum and buffer position in memory before exiting
237  inChksum = tmpChksum;
238  inBufLen = tmpBufLen;
239  isESC = tmpIsESC;
240 }
