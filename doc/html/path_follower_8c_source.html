110 // compute squared length of a vector
111 static inline float sqLen(const FPoint_t vec) {
112  return square(vec.x) + square(vec.y);
113 }

116 // compute the location of a point on a line in the relative distance l (w.r.t.
117 // length of vector vec) where the line is determined by its start point start
118 // and the direction vector vec.
119 static inline FPoint_t pointVec(FPoint_t vec, float l, Point_t start) {
120  // compute FPoint_t p = { (vec.x * l) + start.x, (vec.y * l) + start.y };
121  FPoint_t p = { fmaf(vec.x, l, start.x), fmaf(vec.y, l, start.y) };
122  return p;
123 }

126 // similar to sqLen(pointVec(vec, l, start))
127 static inline float sqLenPointVecF(FPoint_t vec, float l, FPoint_t start) {
128  return square(fmaf(vec.x, l, start.x)) + square(fmaf(vec.y, l, start.y));

142 // Set a path of length pathLen from array pathPoints.
143 // This function is used by commPathFollower() when command FOLLOWER_CMD_NEWPATH
144 // is received from HWPCS as well as by pathFollower_setNewPath().
145 // It checks if the path definition contains at least two points,
146 // resets all path follower local variables and copies the path to a newly
147 // allocated memory region.
148 // Furthermore, it extrapolates an additional point at the end of the path
149 // in direction of the last path segment. The length of the extrapolated segment
150 // is ten times the lookahead distance.
151 static bool setPath(const uint8_t pathLen, const Point_t* pathPoints) {
152  if (pathLen < 2)
153  return false; // path is too short
154 
155  // reset everything to initial condition
156  pathLength = 0;
157  resetState();
158 
159  // if a path was previously received, free its memory
160  if (path != NULL)
161  free(path);
162 
163  // allocate memory for the new path
164  // include space for the extrapolated point at the end
165  size_t s = sizeof(Point_t) * (pathLen + 1);
166  path = (Point_t*)malloc(s);
167  if (path == NULL)
168  return false; // error during memory allocation
169 
170  // copy the path points to the allocated region
171  memcpy(path, pathPoints, s);
172  pathLength = pathLen;
173 
174  // extrapolate a new point in direction of the last segment with a distance
175  // of ten times the lookahead distance from the last point.
176  Point_t
177  *pathPtr = &path[pathLength - 2],
178  lastSegStart = *pathPtr, // equals path[pathLength - 2]
179  lastSegEnd = *++pathPtr; // equals path[pathLength - 1], i.e. the last point
180 
181  // Compute segment vector as (end point - start point)
182  FPoint_t
183  lastSegVec = { lastSegEnd.x - lastSegStart.x, lastSegEnd.y - lastSegStart.y };
184  float
185  // length of last segment
186  lastSegLen = hypotf(lastSegVec.x, lastSegVec.y),
187  // relative distance of extrapolated point (w.r.t. to segment length)
188  relDist = 10.0f * pathFollower_lookaheadDistance / lastSegLen;
189 
190  // extrapolate and append to array path (i.e. set path[pathLength])
191  (++pathPtr)->x = (int16_t)(lastSegVec.x * relDist) + lastSegEnd.x;
192  pathPtr->y = (int16_t)(lastSegVec.y * relDist) + lastSegEnd.y;
193 
194  return true;
195 }

200 // communication callback function for channel CH_IN_PATH_FOLLOW_CTRL in order to control the
201 // path follower from HWPCS
202 void commPathFollower(const uint8_t* packet, __attribute__((unused)) const uint16_t size) {
203  PathFollowerControl_t* ctrl = (PathFollowerControl_t*)packet;
204 
205  if (ctrl->cmd == FOLLOWER_CMD_NEWPATH) { // if a new path was received from HWPCS
206 
207  if (! setPath(ctrl->pathLength, ctrl->points)) {
208  pathFollowerErrorP("Follower: out of memory or path has too few points");
209  } else {
210  pathFollowerInfoP("Path of %" PRIu8 " points, 1.(%" PRIi16 ",%" PRIi16 ") 2.(%" PRIi16 ",%" PRIi16 ") ...", pathLength, path[0].x, path[0].y, path[1].x, path[1].y);
211  }
212 
213  } else if (ctrl->cmd == FOLLOWER_CMD_START) { // start command received
214  if (pathLength == 0) {
215  pathFollowerErrorP("No path available");
216  } else {
217  pathFollower_status.enabled = true;
218  pathFollowerInfoPNoArg("Follower started");
219  }
220  } else if (ctrl->cmd == FOLLOWER_CMD_PAUSE) { // pause command received
221  pathFollower_status.enabled = false;
222  pathFollowerInfoPNoArg("Follower paused");
223  } else if (ctrl->cmd == FOLLOWER_CMD_RESET) { // reset command received
224  resetState();
225  pathFollowerInfoPNoArg("Follower reset");
226  }
227 
228  if (pathFollower_callback != 0)
229  pathFollower_callback(ctrl->cmd, ctrl->pathLength);
230 }

287 // compute a new lookahead point, call this function periodically from main loop
288 bool pathFollower_update(const Pose_t* currentPose) {
289  // exit if following is disabled
290  if (! pathFollower_status.enabled)
291  return false;
292 
293  // copy current position into local variable, allows for large code optimization
294  FPoint_t currentPos = { currentPose->x, currentPose->y };
295 
296  // Step 0: Initialization
297 
298  // if this is the first time, pathFollower_update() was called after either
299  // reset or a path has been set
300  if (! initialized) {
301  initialized = true;
302 
303  // initialize all local and global variables used by the path follower:
304  // In most of the cases (when the distance between the current position and the first
305  // path point path[0] is sufficiently large), the first (current) segment is defined
306  // to go from the current position of the robot to the first path point.
307  // The next segment connects point path[0] to point path[1].
308  pathFollower_status.segEnd = path[0];
309  curSegVec.x = pathFollower_status.segEnd.x - currentPos.x;
310  curSegVec.y = pathFollower_status.segEnd.y - currentPos.y;
311  curSegLenSq = sqLen(curSegVec);
312 
313  // if the distance between current position and first path point is very small
314  if (curSegLenSq < 1.0e-2) {
315  // then already switch to segment between path[0] and path[1]
316  nextSegEnd = path[2];
317  pathFollower_status.segEnd = path[1];
318  pathFollower_status.segStart = path[0];
319  curSegVec.x = pathFollower_status.segEnd.x - pathFollower_status.segStart.x;
320  curSegVec.y = pathFollower_status.segEnd.y - pathFollower_status.segStart.y;
321 
322  curSegLenSq = sqLen(curSegVec);
323  curSeg = 0;
324  } else {
325  // otherwise (most of the cases), we introduce a linear segment from the
326  // current position to the first path point path[0], keeping curSeg = -1
327  nextSegEnd = path[1];
328  pathFollower_status.segStart.x = currentPos.x;
329  pathFollower_status.segStart.y = currentPos.y;
330  }
331 
332  nextSegVec.x = nextSegEnd.x - pathFollower_status.segEnd.x;
333  nextSegVec.y = nextSegEnd.y - pathFollower_status.segEnd.y;
334  nextSegLenSq = sqLen(nextSegVec);
335  nextSegLen = sqrt(nextSegLenSq);
336 
337  curSegLen = sqrt(curSegLenSq);
338 
339  relL = pathFollower_lookaheadDistance / curSegLen;
340  }
341 
342  // Step 1: Find the segment with smallest (squared) distance between current
343  // position and orthogonal projection
344 
345  // compute vector ECur from start of current segment to current position
346  FPoint_t ECur = { currentPos.x - pathFollower_status.segStart.x, currentPos.y - pathFollower_status.segStart.y };
347 
348  float
349  // compute dot product between vector ECur and vector P of current segment
350  // equivalent to P_dot_ECur = curSegVec.x * ECur.x + curSegVec.y * ECur.y, but more accurate
351  P_dot_ECur = fmaf(curSegVec.x, ECur.x, curSegVec.y * ECur.y),
352 
353  // compute relative distance lCur (w.r.t. to current segment length) of
354  // the orthogonal projection of the current position onto the current segment
355  lCur = P_dot_ECur / curSegLenSq,
356 
357  // compute squared distance between current position and orthogonal projection onto current segment
358  // Note, that the following is short for:
359  // 1. compute point OP of orthogonal projection: OP = pointVec(curSegVec, lCur, pathFollower_status.segStart)
360  // 2. compute vector between OP and current position: vecOP_CP = OP - currentPos
361  // 3. compute squared distance between OP and current position: sqLen(vecOP_CP)
362  // and initialize minDist, holding the minimum squared distance between an
363  // orthogonal projection onto a path segment
364  minDist = sqLenPointVecF(curSegVec, -lCur, ECur);
365 
366  // compute vector ENext from start of next segment to current position
367  // Note that the following is equal to:
368  // FPoint_t ENext = { currentPos.x - pathFollower_status.segEnd.x, currentPos.y - pathFollower_status.segEnd.y };
369  // but produces smaller code
370  FPoint_t ENext = { ECur.x - curSegVec.x, ECur.y - curSegVec.y };
371 
372  float
373  // compute dot product between vector ENext and vector P of next segment
374  // equivalent to P_dot_ENext = nextSegVec.x * ENext.x + nextSegVec.y * ENext.y, but more accurate
375  P_dot_ENext = fmaf(nextSegVec.x, ENext.x, nextSegVec.y * ENext.y),
376 
377  // compute relative distance lNext (w.r.t. to next segment length) of
378  // the orthogonal projection of the current position onto the next segment
379  lNext = P_dot_ENext / nextSegLenSq;
380 
381  //communication_log(LEVEL_FINE, "%.3f %.3f", lCur, lNext);
382 
383  // initialize minSeg with the current segment index
384  // minSeg holds the segment index with smallest squared distance minDist
385  int16_t minSeg = curSeg;
386 
387 
388  // if the orthogonal projection of the current position onto the next segment
389  // is located on that segment
390  if ((lNext >= 0.0f) && (lNext <= 1.0f)) {
391  // compute squared distance between current position and orthogonal projection onto next segment
392  float sqDistOPCPNext = sqLenPointVecF(nextSegVec, -lNext, ENext);
393 
394  // if that distance is smaller than the current minimum distance
395  // or if orthogonal projection onto the current segment is already
396  // past the current segment (in which case we definitively need to
397  // switch at least to the next segment)
398  if ((sqDistOPCPNext < minDist) || (lCur > 1.0f) || (minSeg == -1)) {
399  // update minDist and minSeg
400  minDist = sqDistOPCPNext;
401  ++minSeg;
402  }
403  }
404 
405  // now iterate over the remaining segments up to pathFollower_segmentLimit
406  // and try to find orthogonal projections onto these segments with even
407  // smaller minimum distance to the current position.
408 
409  // segStart holds the start of the segment considered
410  Point_t segStart = nextSegEnd;
411  // seg is the segment index (start point of segment path[seg])
412  int16_t seg = curSeg + 1;
413  // pointer in array path for reading out the segment's end point
414  Point_t *pathPtr = &path[seg + 1];
415 
416  // counter for comparison with pathFollower_segmentLimit
417  // initialization with zero means that we covered already the check of the
418  // segment defined by nextSegVec and nextSegEnd from before
419  uint8_t i = 0;
420  while ((++i < pathFollower_segmentLimit) && (++seg < pathLength)) {
421  // get end of segment
422  Point_t segEnd = *++pathPtr;
423 
424  // compute segment vector as (end point - start point)
425  FPoint_t vec = { segEnd.x - segStart.x, segEnd.y - segStart.y };
426 
427  // compute vector E from start of segment to current position
428  FPoint_t E = { currentPos.x - segStart.x, currentPos.y - segStart.y };
429  float
430  // compute dot product between vector E and vector P of segment
431  // equivalent to P_dot_E = vec.x * E.x + vec.y * E.y, but more accurate
432  P_dot_E = fmaf(vec.x, E.x, vec.y * E.y),
433  // compute relative distance l (w.r.t. to segment length) of
434  // the orthogonal projection of the current position onto the segment
435  l = P_dot_E / sqLen(vec);
436 
437  // if the orthogonal projection of the current position onto the segment
438  // is located on that segment
439  if ((l >= 0.0f) && (l <= 1.0f)) {
440  // compute squared distance between current position and orthogonal projection onto segment
441  float sqDistOPCP = sqLenPointVecF(vec, -l, E);
442 
443  // if that distance is smaller than the current minimum distance
444  if (sqDistOPCP < minDist) {
445  // update minDist and minSeg
446  minDist = sqDistOPCP;
447  minSeg = seg;
448  }
449 
450  }
451 
452  // segEnd becomes the start of the segment in the next iteration
453  segStart = segEnd;
454  }
455 
456  // if minSeg hit the maximum, output an error (note that this should not happen)
457  if (minSeg == pathLength) {
458  pathFollowerErrorP("Unable to locate segment for orthogonal projection");
459 
460  // if segment with smallest distance between orthogonal projection and current position
461  // is after the current segment
462  } else if (minSeg > curSeg) {
463  // if the end of the path has been reached, deactivate follower and report success
464  if (minSeg == pathLength - 1) {
465  pathFollower_status.enabled = false;
466  pathFollowerInfoPNoArg("End of path reached");
467  return false;
468  }
469 
470  pathFollowerInfo("Switching from segment %" PRIi16 " to %" PRIi16, curSeg, minSeg);
471 
472  // get a pointer of path[minSeg], i.e. start of segment.
473  // This produces more effective code than just addressing the array multiple
474  // times at consecutive locations.
475  Point_t* pathPtr = &path[minSeg];
476 
477  // if a switch to the next segment is needed, some required variables
478  // are stored already in local variables
479  if (minSeg == curSeg + 1) {
480  ++pathPtr; // advance by one, since access to path[minSeg + 2] is required after this block
481  lCur = lNext;
482  curSegLenSq = nextSegLenSq;
483  curSegVec = nextSegVec;
484  pathFollower_status.segStart = pathFollower_status.segEnd;
485  pathFollower_status.segEnd = nextSegEnd;
486  curSegLen = nextSegLen;
487  // if a switch to past the next segment is needed, update the local variables directly
488  } else {
489  pathFollower_status.segStart = *(pathPtr++); // equals path[minSeg]
490  pathFollower_status.segEnd = *pathPtr; // equals path[minSeg + 1]
491  curSegVec.x = pathFollower_status.segEnd.x - pathFollower_status.segStart.x;
492  curSegVec.y = pathFollower_status.segEnd.y - pathFollower_status.segStart.y;
493  curSegLenSq = sqLen(curSegVec);
494 
495  // recomputation of lCur is required, as it is needed for the next step
496  FPoint_t ECur = { currentPos.x - pathFollower_status.segStart.x, currentPos.y - pathFollower_status.segStart.y };
497  float
498  // equivalent to P_dot_ECur = curSegVec.x * ECur.x + curSegVec.y * ECur.y, but more accurate
499  P_dot_ECur = fmaf(curSegVec.x, ECur.x, curSegVec.y * ECur.y);
500  lCur = P_dot_ECur / curSegLenSq;
501 
502  curSegLen = sqrt(curSegLenSq);
503  }
504 
505  // update the remaining local variables
506  relL = pathFollower_lookaheadDistance / curSegLen;
507 
508  nextSegEnd = *(++pathPtr); // equals path[minSeg + 2]
509  nextSegVec.x = nextSegEnd.x - pathFollower_status.segEnd.x;
510  nextSegVec.y = nextSegEnd.y - pathFollower_status.segEnd.y;
511  nextSegLenSq = sqLen(nextSegVec);
512  nextSegLen = sqrt(nextSegLenSq);
513 
514  curSeg = minSeg;
515  }
516 
517 
518  if (curSeg == pathLength - 2) {
519  pathFollower_extStatus.onLastSegment = true;
520  pathFollower_extStatus.distanceOPToEnd = (1.0f - lCur) * curSegLen;
521  }
522 
523  // Step 2: Locate the lookahead point in distance pathFollower_lookaheadDistance
524  // after the orthogonal projection from step 1
525 
526  // safe-guard in case robot deviates from path too much such that a switch to another segment
527  // does not occur. In this case, the orthogonal projection stops at the end of the current segment.
528  // This prevents the lookahead point from being placed outside the path.
529  if (lCur > 1.0f)
530  lCur = 1.0f;
531 
532  // compute the relative distance of the lookahead point in terms of the current segment
533  float relLA = relL + lCur;
534 
535  // if lookahead point is on the current segment
536  if (relLA <= 1.0f) {
537 
538  // compute lookahead point on current segment
539  pathFollower_status.lookahead = pointVec(curSegVec, relLA, pathFollower_status.segStart);
540 
541  // if lookahead point is after the current segment
542  } else {
543 
544  // subtract the distance between orthogonal projection on current segment to current segment end
545  // from pathFollower_lookaheadDistance, such that Lrem contains the remaining distance of the lookahead
546  // point along the path after the current segment.
547  // This is expressed shorter as:
548  float Lrem = (relLA - 1.0f) * curSegLen;
549 
550  // if lookahead point is on the next segment
551  if (Lrem <= nextSegLen) {
552  // compute lookahead point on next segment
553  pathFollower_status.lookahead = pointVec(nextSegVec, Lrem / nextSegLen, pathFollower_status.segEnd);
554 
555  // if lookahead point is after the next segment
556  } else {
557  // update Lrem by subtracting the length of the next segment
558  Lrem -= nextSegLen;
559 
560  // search for segment where lookahead point is located
561 
562  // segStart holds the start of the segment considered
563  Point_t segStart = nextSegEnd;
564  // seg is the segment index (start point of segment path[seg])
565  uint8_t seg = curSeg + 1;
566  // pointer in array path for reading out the segment's end point
567  Point_t *pathPtr = &path[seg + 1];
568 
569  while (++seg < pathLength) {
570  // get end of segment
571  Point_t segEnd = *++pathPtr;
572 
573  // compute segment vector as (end point - start point)
574  FPoint_t vec = { segEnd.x - segStart.x, segEnd.y - segStart.y };
575  // compute length of vec
576  float vecLen = hypotf(vec.x, vec.y);
577 
578  // if lookahead point is on the segment
579  if (vecLen >= Lrem) {
580  // compute lookahead point on that segment
581  pathFollower_status.lookahead = pointVec(vec, Lrem / vecLen, segStart);
582  break;
583  } else
584  // update Lrem by subtracting the length of the segment
585  Lrem -= vecLen;
586 
587  // segEnd becomes the start of the segment in the next iteration
588  segStart = segEnd;
589  }
590 
591  // if last segment was reached, but still the lookahead point could not
592  // be found, output an error message.
593  // Note that this condition should not occur, except when enlarging
594  // the lookahead distance by more than that distance during a following
595  // approach
596  if (seg == pathLength) {
597  pathFollowerErrorP("Unable to locate segment for lookahead point");
598  }
599  }
600  }
601 
602  // report that pathFollower_status was updated with a new lookahead point
603  // and possibly also new segments start and end points
604  // A controller should now use the lookahead point as an intermediate goal
605  // position and generate appropriate control inputs to the robot
606  return true;
607 }
