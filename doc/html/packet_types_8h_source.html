231 /**
232  * Representation of labyrinth walls.
233  * Each bit in walls represents the state of a wall. The bit number (i.e. wall number)
234  * can be determined with labyrinth_getWallIndex().
235  *
236  * The first (#LABYRINTH_COLS + 1) * #LABYRINTH_ROWS bits represent Western and
237  * Eastern walls of the cells in row0, row1, ...
238  *
239  * The last (#LABYRINTH_ROWS + 1) * #LABYRINTH_COLS bits represent Northern and
240  * Southern walls of the cells in col0, col1, ...
241  *
242  * Use this for communicating the wall state to HWPCS for visualization:
243  * - sent on channel #CH_OUT_LABY_WALLS (0x04)
244  * - size: 16 Bytes (for a 7x7 labyrinth)
245  * - minimum transmission duration (incl. minimum overhead of 5 bytes): 0.42ms
246  */
247 typedef struct __attribute__((__packed__)) {
248  uint8_t cols; ///< number of columns in labyrinth
249  uint8_t rows; ///< number of rows in labyrinth
250  /**
251  * bit representation of walls (14 Bytes for a 7x7 labyrinth).
252  * Computation of the array size is done by a ceiling operation expressed by integers:
253  * - number of Eastern/Western walls is: (cols + 1) * rows
254  * - number of Northern/Southern walls is: (rows + 1) * cols
255  * - sum of both equals total number of walls, i.e. number of bits
256  * - array size is ceiling of number of bits divided by 8
257  * - ceiling of an integer division x/y is achieved by 1 + (x - 1) / y
258  */
259  uint8_t walls[1 + (((LABYRINTH_COLS+1)*LABYRINTH_ROWS + (LABYRINTH_ROWS+1)*LABYRINTH_COLS - 1) >> 3)];
260 } LabyrinthWalls_t;

276 /**
277  * Get pose from HWPCS.
278  * Send request to get pose from HWPCS of tracked AprilTags (visual markers).
279  * If #APRIL_TAG_MAIN is requested, HWPCS will respond with Pose_t on channel
280  * #CH_IN_POSE (0x05).
281  * If #APRIL_TAG_ADDITIONAL is requested, HWPCS will respond with Pose_t on channel
282  * #CH_IN_ADDITIONAL_POSE (0x08).
283  *
284  * A pose request of #APRIL_TAG_MAIN is left unanswered by HWPCS if multiple request
285  * arrive within the interval specified by "Minimum Pose Update Interval" (see
286  * HWPCS -> Preferences -> Scene -> Tracking). Requests for #APRIL_TAG_ADDITIONAL are
287  * unconstrained.
288  *
289  * Accurate information is provided only if the robot stands still and enough time
290  * has passed for HWPCS to acquire the tracking information.
291  * Timing depends on various factors including WLAN. Safe timeouts are
292  * considered at about 500ms or larger.
293  *
294  * - sent on channel #CH_OUT_GET_POSE (0x05)
295  * - size: 1 Byte
296  * - minimum transmission duration (incl. minimum overhead of 5 bytes): 0.12ms
297  */

299  AprilTagType_t aprilTagType; ///< type of AprilTag for which tracking information is requested

338  * Send this packet on channel #CH_OUT_LABY_CELL_INFO (0x08) to display the
339  * given information in the Scene view of HWPCS. This can be useful for
